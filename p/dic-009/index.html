#path settings
VSRC_PATH=vsrc
DUT_PATH=dut
TOP_MODULE=top_module_name
BUILD_PATH=build
UVM_VERBOSITY=MEDIUM

#source file
DUT_FILES=$(wildcard $(VSRC_PATH)/$(DUT_PATH)/*.v $(VSRC_PATH)/$(DUT_PATH)/**/*.v $(VSRC_PATH)/$(DUT_PATH)/*.sv $(VSRC_PATH)/$(DUT_PATH)/**/*.sv)
TOP_FILES=$(wildcard $(VSRC_PATH)/$(TOP_MODULE).v $(VSRC_PATH)/**/$(TOP_MODULE).v $(VSRC_PATH)/$(TOP_MODULE).sv $(VSRC_PATH)/**/$(TOP_MODULE).sv)
SOURCE_FILE=$(DUT_FILES) $(TOP_FILES)

.PHONY: filelist vcs verdi clean all

#compile command 
VCS=vcs +v2k -timescale=1ns/1ps \
		-sverilog \
		-full64	\
		-R \
		-ntb_opts uvm \
		-f ../$(VSRC_PATH)/filelist.f \
		+mindelays \
		-negdelay \
		+neg_tchk \
		-l vcs.log \
		+UVM_VERBOSITY=$(UVM_VERBOSITY)

VERDI=Verdi-Ultra -ssf $(TOP_MODULE).fsdb \
		-nologo \
		-l verdi.log             

all: filelist vcs verdi

#filelist generation
filelist:
	@rm -f $(VSRC_PATH)/filelist.f
	@for dir in $(SOURCE_FILE); do \
		realpath -s "$$(dirname "$$dir")"; \
	done | sort -u | while read dir; do \
		echo "+incdir+$$dir"; \
	done >> $(VSRC_PATH)/filelist.f
	@for file in $(SOURCE_FILE); do \
		echo $$(realpath "$$file") >> $(VSRC_PATH)/filelist.f; \
	done

#start compile and simulate
vcs: filelist
	@mkdir -p $(BUILD_PATH)
	(cd $(BUILD_PATH) && $(VCS))

#run verdi
verdi: filelist
	(cd $(BUILD_PATH) && $(VERDI))
 
#clean
clean:
	rm -rf  $(BUILD_PATH) $(VSRC_PATH)/filelist.f

以下，对该Makefile进行简单说明：

变量定义

首先，我们定义了VSRC_PATH这一变量，用于指定保存所有RTL代码的路径。同时，由于VCS最主要的功能是对设计进行功能仿真验证，所以在VSRC_PATH路径下，通常既会有被测设计（Design Under Test, DUT）的RTL代码，又会有验证环境的RTL代码，那我们是否需要对其进行区分呢？

答案是肯定的，在现代IC验证方法学（如UVM等）中，会明确划分被测设计与验证环境的边界，一个好的习惯是将被测设计的RTL代码放在独立路径下，且在此路径下不包含任何用于验证设计的代码，故此处我们额外定义DUT_PATH这一变量，用于指定被测设计代码所在的路径，实现被测设计与验证环境的解耦。

随后，我们定义了TOP_MODULE这一变量，其用于指定整个环境的顶层模块，该模块通常属于验证环境（因为我们会在验证环境中例化被测设计）。此处我们默认该模块名与该模块所在的文件名相同，比如顶层模块名为tb_dut，则其所在的文件名就应为tb_dut.v或tb_dut.sv，具体原因将在后文说明。

最后，我们定义了BUILD_PATH这一变量，用于为使用VCS时输出的中间文件指定一个独立的保存路径，防止仿真输出污染项目的文件结构。

另外，我们也可以额外定义一些变量用于配置验证环境（可选），比如此处我们定义了UVM_VERBOSITY这一变量，当使用UVM进行验证时，该变量可以全局指定UVM打印信息的冗余度阈值。

设计文件解析

我们使用Linux Bash中的wildcard实现对设计文件的解析，此处我们分别定义了DUT_FILES与TOP_FILES两个变量，其中DUT_FILES代表被测设计的所有RTL代码文件，故我们可在$(VSRC_PATH)/$(DUT_PATH)路径下使用通配符递归匹配所有.v与.sv文件1。

1

DUT_FILES=$(wildcard $(VSRC_PATH)/$(DUT_PATH)/*.v $(VSRC_PATH)/$(DUT_PATH)/**/*.v $(VSRC_PATH)/$(DUT_PATH)/*.sv $(VSRC_PATH)/$(DUT_PATH)/**/*.sv)

TOP_FILES则代表顶层模块所在的文件，我们按与DUT_FILES同样的方法在$(VSRC_PATH)路径下匹配名称为$(TOP_MODULE).v或$(TOP_MODULE).sv的文件，此处即呼应了前文要求顶层模块与其所在文件同名的要求。

随后，我们定义SOURCE_FILE这一变量包含DUT_FILES与TOP_FILES两个变量，代表所有设计文件。我们之所以要独立定义TOP_FILES这一变量，再将其包含进SOURCE_FILE，是因为我们希望保证TOP_FILES排在SOURCE_FILE文件列表的最后，这样在使用VCS解析设计文件时，顶层模块最后才被解析，这可以一定程度上避免解析时出现模块/类定义的依赖问题。

但是，此时会存在一个问题，目前SOURCE_FILE中仅包含了被测设计与顶层模块对应的文件，但对一个完整的验证环境而言，通常远不止顶层模块所在的这一个文件，那么剩余用于构建验证环境的文件应当如何被解析呢？此处我们规定：剩余文件通过在TOP_FILES内部调用include宏包含进验证环境中，这么做有两点原因：

  • 在验证环境中，除TOP_FILES外的其他文件通常是对验证环境的激励、参考模型与计分板等组件的类定义，在使用VCS时，这些文件通常需要指定明确的解析顺序，否则就会产生定义依赖问题，因此，在TOP_FILES中手动通过include宏按顺序包含这些文件，是一个相对稳定的方案。
  • 对一个成熟的验证环境（如UVM）而言，其结构应当是固定的，因此，我们虽然在TOP_FILES中手动通过include宏包含了剩余文件，但在被测设计更改时，我们并不需要修改验证环境的结构，进而也不必重复修改这些用于构建验证环境的宏定义。

构建规则

首先，我们分别定义VCS与VERDI两个变量，用于指定在命令行中运行VCS与Verdi时的参数配置：对VCS而言，我们使用filelist.f导入所有设计文件，该文件通过filelist目标构建；对Verdi而言，我们会在启动的同时读入仿真生成的$(TOP_MODULE).fsdb波形文件，因此，如希望使用Verdi，应当在验证环境中加入：

1
2
3
4

initial begin
    $dumpfile("top_module_name.fsdb");
    $dumpvars(0, top_module_name);
end

对filelist目标，我们同样使用Linux Bash指令以在VSRC_PATH路径下生成filelist.f，该段指令生成的filelist.f中，首先会使用+incdir+包含SOURCE_FILE中所有文件的上级绝对路径（使用sort -u去重）供VCS搜索，随后包含SOURCE_FILE中所有文件的绝对路径，比如：

2
3

+incdir+/home/posvirus/UVM/vsrc
+incdir+/home/posvirus/UVM/vsrc/dut
/home/posvirus/UVM/vsrc/dut/dut.sv
/home/posvirus/UVM/vsrc/top_tb.sv

对vcs与verdi目标，其会进入BUILD_PATH分别运行VCS与Verdi，用于仿真验证与查看波形。

对clean目标，其会直接删除BUILD_PATH路径与filelist.f，用于清除所有用于仿真验证的输出文件。

  1. 如果RTL代码文件中还包含以.vh与.svh等为后缀的头文件，可增加通配符的匹配规则以包含更多类型的文件，也可直接在RTL代码文件中使用include宏包含这些头文件。 ↩︎

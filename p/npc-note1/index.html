static int cmd_q(char *args) {
  // if nemu hits bad trap, hold state
  if ((nemu_state.state == NEMU_ABORT) || 
      ((nemu_state.state == NEMU_END) && (nemu_state.halt_ret != 0))) {
    // do nothing
  } else {
    nemu_state.state = NEMU_QUIT;
  }
  return -1;
}

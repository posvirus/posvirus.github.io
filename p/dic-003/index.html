module seq_detector #(
    parameter SEQ     = 8'h5A,
    parameter SEQ_LEN = 8              
) (
    input  wire clk,     // system clock
    input  wire rst_n,   // reset
    input  wire data_in, // serial data input
    output reg  match    // detected signal
);
    // shift register
    reg [SEQ_LEN-1:0] shift_reg;

    always @(posedge clk or negedge rst_n) begin
        if (~rst_n) begin
            shift_reg <= {(SEQ_LEN){~SEQ[SEQ_LEN-1]}}; // diff MSB
        end else if (shift_reg == SEQ[SEQ_LEN-1:0]) begin
            shift_reg <= {(SEQ_LEN){~SEQ[SEQ_LEN-1]}}; // diff MSB
        end else begin
            shift_reg <= {shift_reg[SEQ_LEN-2:0], data_in}; // bit shift
        end
    end

    // compare output
    always @(posedge clk or negedge rst_n) begin
        if (~rst_n) begin
            match <= 1'b0;
        end else if (shift_reg == SEQ[SEQ_LEN-1:0]) begin
            match <= 1'b1;
        end else begin
            match <= 1'b0;
        end
    end

endmodule

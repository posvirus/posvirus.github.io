module clk_divider #(
    parameter DIV_NUM   = 4'd10, // should be set to even
    parameter CNT_WIDTH = 4
) (
    input  wire clk,
    input  wire rst_n,
    output reg  clk_out
);
    localparam PULSE_WIDTH_CYC = DIV_NUM >> 1;

    // cycle counter
    reg [CNT_WIDTH-1:0] cyc_cnt;

    always @(posedge clk or negedge rst_n) begin
        if (~rst_n) begin
            cyc_cnt <= {(CNT_WIDTH){1'b0}};
        end else if (cyc_cnt == DIV_NUM - 1) begin
            cyc_cnt <= {(CNT_WIDTH){1'b0}};
        end else begin
            cyc_cnt <= cyc_cnt + 'b1;
        end
    end

    // output clock
    always @(*) begin
        if (cyc_cnt < PULSE_WIDTH_CYC) begin
            clk_out = 1'b0;
        end else begin
            clk_out = 1'b1;
        end
    end

endmodule

module clk_divider (
    input  wire clk,
    input  wire rst_n,
    output reg  clk_out
);
    localparam FRAC_Q = 10;
    localparam FRAC_K = 5;
    localparam FRAC_ORDER = 10'b0101001010;
    localparam ALC_CNT_WIDTH = 4;
    localparam CYC_CNT_WIDTH = 3;

    // allocation counter
    reg [ALC_CNT_WIDTH-1:0] alc_cnt;
    reg [CYC_CNT_WIDTH-1:0] cyc_cnt;

    always @(posedge clk or negedge rst_n) begin
        if (~rst_n) begin
            alc_cnt <= {(ALC_CNT_WIDTH){1'b0}};
        end else if ((FRAC_ORDER[alc_cnt] == 1'b0) && (cyc_cnt == FRAC_K)) begin
            alc_cnt <= (alc_cnt < FRAC_Q - 1) ? (alc_cnt + 'b1) : ({(ALC_CNT_WIDTH){1'b0}});
        end else if ((FRAC_ORDER[alc_cnt] == 1'b1) && (cyc_cnt == FRAC_K + 1)) begin
            alc_cnt <= (alc_cnt < FRAC_Q - 1) ? (alc_cnt + 'b1) : ({(ALC_CNT_WIDTH){1'b0}});
        end else begin
            alc_cnt <= alc_cnt;
        end
    end

    // cycle counter
    always @(posedge clk or negedge rst_n) begin
        if (~rst_n) begin
            cyc_cnt <= {(CYC_CNT_WIDTH){1'b0}};
        end else if ((FRAC_ORDER[alc_cnt] == 1'b0) && (cyc_cnt == FRAC_K)) begin
            cyc_cnt <= 'b1;
        end else if ((FRAC_ORDER[alc_cnt] == 1'b1) && (cyc_cnt == FRAC_K + 1)) begin
            cyc_cnt <= 'b1;
        end else begin
            cyc_cnt <= cyc_cnt + 'b1;
        end
    end

    // output clock
    always @(*) begin
        if ((FRAC_ORDER[alc_cnt] == 1'b0) && (cyc_cnt <= (FRAC_K >> 1))) begin
            clk_out = 1'b0;
        end else if ((FRAC_ORDER[alc_cnt] == 1'b1) && (cyc_cnt <= ((FRAC_K + 1) >> 1))) begin
            clk_out = 1'b0;
        end else begin
            clk_out = 1'b1;
        end
    end

endmodule

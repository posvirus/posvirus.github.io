Context* __am_irq_handle(Context *c) {
  if (user_handler) {
    Event ev = {0};
    switch (c->mcause) {
      case 11: 
#ifdef __riscv_e

#else
        if (c->gpr[17] == -1) { // $a7
#endif
          ev.event = EVENT_YIELD;

          // software: jump to next instruction
          c->mepc += 4; break;
        } else {
          ev.event = EVENT_SYSCALL; break;
        }
      default: ev.event = EVENT_ERROR; break;
    }

    c = user_handler(ev, c);
    assert(c != NULL);
  }

  return c;
}
